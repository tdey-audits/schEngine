\question \textbf{2.} Determine the ratio in which the point $P(x, 2)$ divides the line segment joining the points $A(12, 5)$ and $B(4, -3)$.

\end{center}

\textbf{Solution:}
\begin{enumerate}

  \item Substitute values and solve for the ratio $m:n$.
\end{enumerate}
\textbf{Derivation:}
\begin{center}
\fitmath{\begin{aligned}
2 = \frac{m(-3) + n(5)}{m+n} \\
2m + 2n = -3m + 5n \\
5m = 3n \\
\frac{m}{n} = \frac{3}{5}
\end{aligned}}
\end{center}

\end{aligned}}
