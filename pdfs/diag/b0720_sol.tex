\question \textbf{1.} A quadrilateral $\text{ABCD}$ is drawn to circumscribe a circle. If $AB = 6$ cm, $BC = 7$ cm, and $CD = 4$ cm, find the length of $AD$.

\textbf{Answer:}
\begin{center}
\fitmath{3 cm}
\end{center}

\textbf{Solution:}
\begin{enumerate}
  \item Use the property that for a circumscribed quadrilateral, the sums of opposite sides are equal.
  \item $AB + CD = BC + AD$.
\end{enumerate}
\textbf{Derivation:}
\begin{center}
\fitmath{\begin{aligned}
6 + 4 = 7 + AD \\
10 = 7 + AD \\
AD = 3 \text{ cm}
\end{aligned}}
\end{center}

\question \textbf{3.} Two concentric circles are of radii $5$ $cm$ and $3$ $cm$. Find the length of the chord of the larger circle which touches the smaller circle.

\textbf{Answer:}
\begin{center}
\fitmath{8 cm}
\end{center}

\textbf{Solution:}
\begin{enumerate}
  \item Let $O$ be the common center. Let $AB$ be the chord of the larger circle tangent to the smaller circle at $M$.
  \item Then $OM \perp AB$. In $\triangle OMA$, $OA = 5$ $cm$ and $OM = 3$ $cm$.
  \item By Pythagoras theorem, $AM = \sqrt{OA^2 - OM^2} = \sqrt{25 - 9} = 4$ $cm$.
  \item Since the perpendicular from the centre bisects the chord, $AB = 2 \times AM = 8$ $cm$.
\end{enumerate}
\textbf{Derivation:}
\begin{center}
\fitmath{\begin{aligned}
AM^2 + OM^2 = OA^2 \\
AM^2 + 3^2 = 5^2 \\
AM^2 = 25 - 9 = 16 \\
AM = 4 \\
AB = 2 \times AM = 8
\end{aligned}}
\end{center}

\question \textbf{4.} If tangents $PA$ and $PB$ from a point $P$ to a circle with centre $O$ are inclined to each other at an angle of $80^\circ$, find $\angle POA$.

\textbf{Answer:}
\begin{center}
\fitmath{50^\circ}
\end{center}

\textbf{Solution:}
\begin{enumerate}
  \item In quadrilateral $\text{OAPB}$, $\angle OAP = \angle OBP = 90^\circ$ (radius $\perp$ tangent).
  \item The sum of angles in a quadrilateral is $360^\circ$, so $\angle AOB = 180^\circ - 80^\circ = 100^\circ$.
  \item Since $OP$ bisects $\angle AOB$, $\angle POA = \frac{1}{2} \angle AOB$.
  \item Therefore, $\angle POA = 100^\circ / 2 = 50^\circ$.
\end{enumerate}
\textbf{Derivation:}
\begin{center}
\fitmath{\begin{aligned}
\angle AOB + \angle APB = 180^\circ \\
\angle AOB + 80^\circ = 180^\circ \\
\angle AOB = 100^\circ \\
\angle POA = \frac{1}{2} \angle AOB = 50^\circ
\end{aligned}}
\end{center}

\question \textbf{5.} A quadrilateral $\text{ABCD}$ is drawn to circumscribe a circle. Prove that $AB + CD = AD + BC$.

\textbf{Answer:}
\begin{center}
\fitmath{\text{Proved}}
\end{center}

\textbf{Solution:}
\begin{enumerate}
  \item Tangents drawn from an external point to a circle are equal in length.
  \item Let the circle touch $AB, BC, CD, DA$ at $P, Q, R, S$ respectively. Then $AP=AS, BP=BQ, CR=CQ, DR=DS$.
  \item Summing these: $(AP+BP) + (CR+DR) = (AS+DS) + (BQ+CQ)$.
  \item This simplifies to $AB + CD = AD + BC$.
\end{enumerate}
\textbf{Derivation:}
\begin{center}
\fitmath{\begin{aligned}
AP=AS, BP=BQ, CR=CQ, DR=DS \\
(AP+BP) + (CR+DR) = (AS+DS) + (BQ+CQ) \\
AB + CD = AD + BC
\end{aligned}}
\end{center}

\question \textbf{6.} Prove that the perpendicular at the point of contact to the tangent to a circle passes through the centre of the circle.

\textbf{Answer:}
\begin{center}
\fitmath{\text{Proved using contradiction}}
\end{center}

\textbf{Solution:}
\begin{enumerate}
  \item Assume that the perpendicular to the tangent at the point of contact $P$ does not pass through the centre $O$.
  \item Let another line through $P$ pass through $O$, such that $\angle OPT = 90^\circ$ (where $T$ is on the tangent).
  \item This contradicts the given that the perpendicular is at $P$.
  \item Hence, the perpendicular must pass through $O$.
\end{enumerate}
\textbf{Derivation:}
\begin{center}
\fitmath{\begin{aligned}
\text{Let } OP \perp \text{tangent at } P. \\
\text{If } O'P \perp \text{tangent at } P, \text{ then } \angle O'PT = \angle OPT = 90^\circ. \\
\text{This implies } O'P \text{ and } OP \text{ coincide.} \\
\therefore O' = O.
\end{aligned}}
\end{center}
