\question \textbf{1.} If a tangent $PA$ and a tangent $PB$ from a point $P$ to a circle with centre $O$ are inclined to each other at an angle of $80^\circ$, then $\angle POA$ is equal to:


\begin{center}
\includegraphics[width=0.48\textwidth]{/home/user/Desktop/backup/schEngine/pdfs/diag/diagrams/diagram_063.pdf}
\end{center}


\textbf{Answer:}
(A)

\textbf{Solution:}
\begin{enumerate}
  \item In $\triangle OAP$ and $\triangle OBP$, $OA=OB$ (radii), $PA=PB$ (tangents from an external point), and $OP=OP$ (common).
  \item Thus, $\triangle OAP \cong \triangle OBP$ by SSS congruence.
  \item Consequently, $\angle POA = \angle POB$. Since $\angle APB = 80^\circ$, we have $\angle OPA = 40^\circ$.
  \item In $\triangle OAP$, $\angle OAP = 90^\circ$ (radius is perpendicular to tangent).
  \item Therefore, $\angle POA = 180^\circ - 90^\circ - 40^\circ = 50^\circ$.
\end{enumerate}
\textbf{Derivation:}
\begin{center}
\fitmath{\begin{aligned}
OP \text{ bisects } \angle APB \\
\Rightarrow \angle OPA = \frac{80^\circ}{2} = 40^\circ \\
\triangle OAP \text{ is right-angled at } A \\
\angle POA + \angle OPA + \angle OAP = 180^\circ \\
\angle POA + 40^\circ + 90^\circ = 180^\circ \\
\angle POA = 180^\circ - 130^\circ = 50^\circ
\end{aligned}}
\end{center}

\question \textbf{2.} A circle touches all four sides of a quadrilateral $\text{ABCD}$. If $AB = 6 \text{ cm}$, $BC = 7 \text{ cm}$, and $CD = 4 \text{ cm}$, then the length of $AD$ is:

\textbf{Answer:}
(C)

\textbf{Solution:}
\begin{enumerate}
  \item For a quadrilateral circumscribing a circle, the sum of opposite sides are equal.
  \item This is based on the theorem that lengths of tangents drawn from an external point to a circle are equal.
  \item So, $AB + CD = BC + AD$.
  \item Substitute the given values: $6 + 4 = 7 + AD$.
  \item Solve for $AD$: $10 = 7 + AD \Rightarrow AD = 3 \text{ cm}$.
\end{enumerate}
\textbf{Derivation:}
\begin{center}
\fitmath{\begin{aligned}
AB + CD = BC + AD \\
6 + 4 = 7 + AD \\
10 = 7 + AD \\
AD = 10 - 7 = 3
\end{aligned}}
\end{center}

\end{aligned}}

\question \textbf{21.} Prove that the tangents drawn at the ends of a diameter of a circle are parallel.

\end{enumerate}
\textbf{Derivation:}
\begin{center}
\fitmath{\begin{aligned}
\angle OAP = 90^\circ \\
\angle OBS = 90^\circ \\
\angle OAP = \angle OBS \\
PQ \parallel RS
\end{aligned}}
\end{center}

\question \textbf{22.} A quadrilateral $\text{ABCD}$ is drawn to circumscribe a circle. Prove that $AB + CD = AD + BC$.

\textbf{Answer:}
\begin{center}
\fitmath{\text{Using the property that lengths of tangents drawn from an external point to a circle are equal}.}
\end{center}

\textbf{Solution:}
\begin{enumerate}
  \item Let the circle touch sides $AB, BC, CD, DA$ at $P, Q, R, S$ respectively.
  \item Since tangents from an external point are equal, $AP=AS, BP=BQ, CR=CQ, DR=DS$.
  \item Summing these: $(AP+BP) + (CR+DR) = (AS+DS) + (BQ+CQ)$.
  \item This simplifies to $AB + CD = AD + BC$.
\end{enumerate}
\textbf{Derivation:}
\begin{center}
\fitmath{\begin{aligned}
AP=AS, BP=BQ, CR=CQ, DR=DS \\
(AP+BP)+(CR+DR) = (AS+DS)+(BQ+CQ) \\
AB+CD = AD+BC
\end{aligned}}
\end{center}

\question \textbf{23.} Prove that the lengths of tangents drawn from an external point to a circle are equal. Using this result, find the length of $PQ$ if $PQ$ is a tangent to a circle with centre $O$ and radius $5 \text{ cm}$, where $OQ = 13 \text{ cm}$ and $P$ is the external point.

\begin{center}
\fitmath{12 \text{ cm}}
\end{center}

\begin{enumerate}
  \item State the theorem: Tangents from an external point are equal.
  \item Construct $\triangle OPQ$ where $\angle OQP = 90^\circ$ because the radius is perpendicular to the tangent at the point of contact.
  \item Apply Pythagoras theorem in $\triangle OPQ$: $OP^2 = OQ^2 + PQ^2$.

\question \textbf{24.} A quadrilateral $\text{ABCD}$ is drawn to circumscribe a circle. Prove that $AB + CD = AD + BC$.

  \item Write equations: $AP=AS$, $BP=BQ$, $CQ=CR$, and $DR=DS$.
  \item Sum the left sides and right sides: $(AP+BP) + (CR+DR) = (AS+DS) + (BQ+CQ)$.
  \item Observe that $AP+BP=AB$, $CR+DR=CD$, $AS+DS=AD$, and $BQ+CQ=BC$.
  \item Conclude that $AB + CD = AD + BC$.
\end{enumerate}
\textbf{Derivation:}
\begin{center}
\fitmath{\begin{aligned}
AP = AS, BP = BQ, CR = CQ, DR = DS \\
(AP + BP) + (CR + DR) = (AS + DS) + (BQ + CQ) \\
AB + CD = AD + BC
\end{aligned}}
\end{center}

OA^2 = OM^2 + AM^2 \\
5^2 = 3^2 + AM^2 \\
25 = 9 + AM^2 \\
AM = 4 \\

\end{aligned}}
