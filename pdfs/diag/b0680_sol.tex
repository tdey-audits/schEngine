\question \textbf{1.} The total surface area of a cube whose edge is $5 \text{ cm}$ is:

\textbf{Answer:}
(A)

\textbf{Solution:}

\question \textbf{4.} Assertion (A): The total surface area of a solid hemisphere of radius $r$ is $3\pi r^2$. Reason (R): The total surface area of a sphere is $4\pi r^2$.

\textbf{Answer:}
(B)

\textbf{Solution:}
\begin{enumerate}
  \item The total surface area of a hemisphere includes the curved surface area ($2\pi r^2$) and the area of the circular base ($\pi r^2$). Thus, TSA = $3\pi r^2$. Assertion is true.
  \item The formula for the total surface area of a sphere is indeed $4\pi r^2$. Reason is true.
  \item The reason explains the surface area of a sphere, not the derivation of the hemisphere's area. Thus, R is not the correct explanation for A.
\end{enumerate}
\textbf{Derivation:}
\begin{center}
\fitmath{TSA_{\text{hemisphere}} = 2\pi r^2 + \pi r^2 = 3\pi r^2}
\end{center}
\hfill \textit{Surface Areas and Volumes — Surface Area of Solids}

\question \textbf{5.} Assertion (A): If the radius of a cylinder is doubled and its height is halved, the volume remains the same. Reason (R): The volume of a cylinder is given by $V = \pi r^2 h$.

\textbf{Answer:}
(D)

\textbf{Solution:}
\begin{enumerate}
  \item Let initial volume be $V = \pi r^2 h$.
  \item New volume $V' = \pi (2r)^2 (h/2) = \pi (4r^2) (h/2) = 2\pi r^2 h = 2V$. Assertion is false.
  \item The volume formula for a cylinder is correctly identified as $\pi r^2 h$. Reason is true.
\end{enumerate}
\textbf{Derivation:}
\begin{center}
\fitmath{V_{new} = \pi (2r)^2 \times \frac{h}{2} = 2\pi r^2 h}
\end{center}
\hfill \textit{Surface Areas and Volumes — Volume of Cylinder}

\question \textbf{7.} Assertion (A): The slant height $l$ of a cone is given by $l = \sqrt{h^2 + r^2}$. Reason (R): This is derived from Pythagoras theorem applied to the triangle formed by height, radius, and slant height.

\textbf{Answer:}
(A)

\textbf{Solution:}
\begin{enumerate}
  \item The slant height $l$, radius $r$, and vertical height $h$ form a right-angled triangle in a cone.
  \item By Pythagoras theorem, the hypotenuse $l^2 = r^2 + h^2$, so $l = \sqrt{r^2 + h^2}$. Assertion is true.
  \item The Reason correctly identifies the geometric origin of the formula. Thus, R is the correct explanation.
\end{enumerate}
\textbf{Derivation:}
\begin{center}
\fitmath{\begin{aligned}
l^2 = r^2 + h^2 \\
\implies l = \sqrt{r^2 + h^2}
\end{aligned}}
\end{center}

\question \textbf{8.} Assertion (A): If the surface area of a sphere is $616 \text{ cm}^2$, its radius is $7 \text{ cm}$. Reason (R): The surface area of a sphere is $4\pi r^2$.

\textbf{Answer:}
(A)

\textbf{Solution:}
\begin{enumerate}
  \item Using $4\pi r^2 = 616$, we have $4 \times (22/7) \times r^2 = 616$.
  \item $r^2 = 616 \times 7 / (4 \times 22) = 616 \times 7 / 88 = 7 \times 7 = 49$.
  \item $r = 7 \text{ cm}$. Assertion is true.
  \item Reason is the correct formula used for the calculation. Thus, R is the correct explanation.
\end{enumerate}
\textbf{Derivation:}
\begin{center}
\fitmath{\begin{aligned}
4 \times \frac{22}{7} \times r^2 = 616 \\
\implies r^2 = 49 \\
\implies r = 7
\end{aligned}}
\end{center}
\hfill \textit{Surface Areas and Volumes — Surface Area of Sphere}

(B)
